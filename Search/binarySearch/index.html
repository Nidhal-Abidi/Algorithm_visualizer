        def binarySearch(array,k,low,high):
  while (low <= high):
    mid = (high + low)//2
        
    if (array[mid] == k):
      return mid
        
    elif (array[mid] < k):
      low = mid + 1
        
    else:
      high = mid - 1
        
  return -1

  #This algo requires the array 
  #to be sorted to work properly
