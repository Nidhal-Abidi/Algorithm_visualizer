  void selectionSort(int array[], int size) {
  for (int step = 0; step < size - 1; step++) {
    int min_idx = step;
    
    for (int i = step + 1; i < size; i++) {
      // Select the minimum element in each loop.
      if (array[i] < array[min_idx]){
        min_idx = i;
      }
    }   
    
    if (min_idx != step){
      swap(&array[min_idx], &array[step]);
    }
  }
}
